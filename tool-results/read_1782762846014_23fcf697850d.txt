    24→/**
    25→ * SpeakerIntroCanvas — the data-driven mockup renderer.
    26→ *
    27→ * Takes a `SpeakerIntroData` object and renders the full Speaker Intro
    28→ * mockup as a 3:2 landscape canvas. Every component is a separate
    29→ * absolutely-positioned div bound to a field in `data`. Edit the JSON
    30→ * in the parent editor and the canvas re-renders live.
    31→ *
    32→ * Canvas size: 1200×800 (3:2). The parent scales it down via CSS
    33→ * transform for the on-screen preview; the underlying DOM stays at
    34→ * 1200×800 so PNG export is print-quality.
    35→ *
    36→ * Editable mode (editable=true):
    37→ *   - Image areas show a dashed blue outline + a "Replace" button.
    38→ *   - Click the button → opens the image picker (onPickImage).
    39→ *   - Drag the image → pans (updates focusX/focusY via onPlacementChange).
    40→ *   - Wheel on the image → zooms (updates zoom).
    41→ *   - Double-click → resets placement to default.
    42→ *
    43→ * Sections-editable mode (sectionsEditable=true):
    44→ *   - Text sections (header, topic, speakers, sponsors, collaborators,
    45→ *     branding, qr, footer) get wrapped in <SectionBox> which makes
    46→ *     them draggable + 8-handle resizeable.
    47→ *   - Layout persists in `data.sectionLayout[id] = { pos, scale }`.
    48→ *   - Alignment guides appear when dragging (cyan lines at canvas edges,
    49→ *     centers, and peer box edges).
    50→ *   - Text sections always render at zIndex >= 50 so they stay above
    51→ *     images and overlays.
    52→ */
    53→
    54→const CANVAS_W = 1200;
    55→const CANVAS_H = 800;
    56→
    57→type Props = {
    58→  data: SpeakerIntroData;

    62→  /** When true, text sections become draggable + resizeable. */
    63→  sectionsEditable?: boolean;
    64→  /** Called when the user clicks "Replace" on an image slot. */
    65→  onPickImage?: (slot: ImageSlot) => void;
    66→  /** Called whenever an image is dragged / zoomed. */
    67→  onPlacementChange?: (slot: ImageSlot, placement: ImagePlacement) => void;
    68→  /** Called when the user drags a resize corner handle on an image. */
    69→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
    70→  /** Called when a section is dragged to a new position. */
    71→  onSectionMove?: (id: SectionId, pos: SectionPos) => void;
    72→  /** Called when a section is resized via a corner/edge handle. */
    73→  onSectionResize?: (id: SectionId, scale: number) => void;
    74→  /** Called when a section is resized via a mid-edge handle — updates the
    75→   *  box's explicit width/height in canvas px. */
    76→  onSectionBoxResize?: (id: SectionId, size: SectionBoxSize) => void;
    77→  /** Called when the hero overlay z-index changes (front/back button). */
    78→  onHeroZChange?: (z: number) => void;
    79→  /** Called when the triangle overlay z-index changes (front/back button). */
    80→  onTriangleZChange?: (z: number) => void;

    82→  onHeroScaleXChange?: (n: number) => void;
    83→  /** Called when the hero overlay Y scale changes (slider). */
    84→  onHeroScaleYChange?: (n: number) => void;

   784→/**
   785→ * EditableImage — wraps a next/image with placement (object-position + scale)
   786→ * and (optionally) edit-mode interactions: click-to-replace, drag-to-pan,
   787→ * wheel-to-zoom, double-click-to-reset.
   788→ *
   789→ * Used for the hero background and the speaker headshots.
   790→ */
   791→function EditableImage({
   792→  slot,
   793→  src,
   794→  alt,
   795→  placement,
   796→  editable,
   797→  previewScale,
   798→  onPickImage,
   799→  onPlacementChange,
   800→  onSizeChange,
   801→  sizeMultiplier,
   802→  sizeLabel,
   803→  containerClass,
   804→  objectFit,
   805→}: {
   806→  slot: ImageSlot;
   807→  src: string;
   808→  alt: string;
   809→  placement?: ImagePlacement;
   810→  editable?: boolean;
   811→  previewScale: number;
   812→  onPickImage?: (slot: ImageSlot) => void;
   813→  onPlacementChange?: (slot: ImageSlot, p: ImagePlacement) => void;
   814→  /** Called when the user drags a resize corner handle. Receives the new
   815→   *  size multiplier (e.g. 1.5 = 150% of default). */
   816→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
   817→  /** Current size multiplier (read from data). Used to seed the resize
   818→   *  drag delta. Defaults to 1. */
   819→  sizeMultiplier?: number;
   820→  /** Small label shown next to the resize readout (e.g. "photo", "logo"). */
   821→  sizeLabel?: string;
   822→  containerClass: string;
   823→  objectFit: "cover" | "contain";
   824→}) {
   825→  const { focusX, focusY, zoom } = resolvePlacement(placement);
   826→  // We track drag state on a ref so we don't re-render on every mousemove.
   827→  const dragRef = useRef<{
   828→    startX: number;
   829→    startY: number;
   830→    startFocusX: number;
   831→    startFocusY: number;
   832→  } | null>(null);
   833→  // Resize drag state — separate from pan drag so they don't conflict.
   834→  const resizeRef = useRef<{
   835→    startX: number;
   836→    startY: number;
   837→    startSize: number;
   838→    /** Which corner is being dragged. The corner determines whether
   839→     *  moving the mouse up-left grows or shrinks the image. */
   840→    corner: "nw" | "ne" | "se" | "sw";
   841→  } | null>(null);
   842→
   843→  function handleMouseDown(e: React.MouseEvent) {
   844→    if (!editable || !onPlacementChange) return;
   845→    // Only start a drag on left-click outside the Replace button.
   846→    if (e.button !== 0) return;
   847→    e.preventDefault();
   848→    dragRef.current = {
   849→      startX: e.clientX,
   850→      startY: e.clientY,
   851→      startFocusX: focusX,
   852→      startFocusY: focusY,
   853→    };
   854→    // Switch to grabbing cursor.
   855→    (e.currentTarget as HTMLElement).style.cursor = "grabbing";
   856→
   857→    const onMove = (ev: MouseEvent) => {
   858→      const d = dragRef.current;
   859→      if (!d) return;
   860→      const dx = ev.clientX - d.startX;
   861→      const dy = ev.clientY - d.startY;
   862→      // Convert screen-pixel delta to % of container.
   863→      // The container's pixel width at screen-scale is:
   864→      //   containerWidthPx (canvas) * previewScale.
   865→      // We don't know containerWidthPx here without measuring it; use a
   866→      // reasonable approximation: 1% per (previewScale * 6) px, which gives
   867→      // ~6 px of drag = 1% on the canvas at scale=1.
   868→      const sensitivity = 6 * previewScale;
   869→      const nextFocusX = Math.max(
   870→        0,
   871→        Math.min(100, d.startFocusX - dx / sensitivity),
   872→      );
   873→      const nextFocusY = Math.max(
   874→        0,
   875→        Math.min(100, d.startFocusY - dy / sensitivity),
   876→      );
   877→      onPlacementChange(slot, {
   878→        focusX: nextFocusX,
   879→        focusY: nextFocusY,
   880→        zoom,
   881→      });
   882→    };
   883→    const onUp = () => {
   884→      dragRef.current = null;
   885→      window.removeEventListener("mousemove", onMove);
   886→      window.removeEventListener("mouseup", onUp);
   887→      // Restore cursor.
   888→      const el = document.getElementById(`editable-img-${slotKey(slot)}`);
   889→      if (el) el.style.cursor = "grab";
   890→    };
   891→    window.addEventListener("mousemove", onMove);
   892→    window.addEventListener("mouseup", onUp);
   893→  }
   894→
   895→  const containerRef = useRef<HTMLDivElement>(null);
   896→    // Attach a NON-PASSIVE wheel listener so preventDefault
   897→    // actually stops the parent workspace from scrolling.
   898→    // React's onWheel is passive by default → preventDefault
   899→    // is a no-op there + logs a console warning.
   900→    useNonPassiveWheel(containerRef, handleWheel, !!editable);
   901→
   902→    function handleWheel(e: WheelEvent) {
   903→    if (!editable || !onPlacementChange) return;
   904→    // preventDefault + stopPropagation are already called by the
   905→    // useNonPassiveWheel hook (non-passive native listener), so
   906→    // the parent workspace does not scroll while the user spins
   907→    // the wheel over a hovered image.
   908→const step = e.deltaY < 0 ? 0.1 : -0.1;
   909→    const nextZoom = Math.max(0.01, zoom + step);
   910→    onPlacementChange(slot, {
   911→      focusX,
   912→      focusY,
   913→      zoom: nextZoom,
   914→    });
   915→  }
   916→
   917→  function handleDoubleClick() {
   918→    if (!editable || !onPlacementChange) return;
   919→    onPlacementChange(slot, { focusX: 50, focusY: 50, zoom: 1 });
   920→  }
   921→
   922→  /**
   923→   * handleResizeMouseDown — starts a resize drag from one of the 4 corner
   924→   * handles. The corner determines the direction:
   925→   *   - SE (bottom-right): drag down-right = grow, up-left = shrink
   926→   *   - NW (top-left):     drag up-left   = grow, down-right = shrink
   927→   *   - NE (top-right):    drag up-right  = grow, down-left = shrink
   928→   *   - SW (bottom-left):  drag down-left = grow, up-right = shrink
   929→   *
   930→   * We use the diagonal distance (dx + dy with appropriate sign) so the
   931→   * resize feels natural regardless of which corner is grabbed.
   932→   */
   933→  function handleResizeMouseDown(
   934→    e: React.MouseEvent,
   935→    corner: "nw" | "ne" | "se" | "sw",
   936→  ) {
   937→    if (!editable || !onSizeChange) return;
   938→    if (e.button !== 0) return;
   939→    e.preventDefault();
   940→    e.stopPropagation(); // don't trigger pan-drag
   941→    const startSize = sizeMultiplier ?? 1;
   942→    resizeRef.current = { startX: e.clientX, startY: e.clientY, startSize, corner };
   943→
   944→    const onMove = (ev: MouseEvent) => {
   945→      const r = resizeRef.current;
   946→      if (!r) return;
   947→      const dx = ev.clientX - r.startX;
   948→      const dy = ev.clientY - r.startY;
   949→      // For each corner, compute the signed diagonal distance so that
   950→      // moving "outward" (away from the image center) increases the size
   951→      // and moving "toward the center" decreases it. The SW and NE
   952→      // formulas were previously inverted — fixed per user spec.
   953→      let signedDiag: number;
   954→      switch (r.corner) {
   955→        case "se": signedDiag = dx + dy; break;            // down-right grows
   956→        case "nw": signedDiag = -(dx + dy); break;          // up-left grows
   957→        case "ne": signedDiag = dx - dy; break;             // up-right grows (dx>0 grows, dy<0 grows)
   958→        case "sw": signedDiag = -dx + dy; break;            // down-left grows (dx<0 grows, dy>0 grows)
   959→      }
   960→      // 100px of drag = 1.0× size change (so dragging 50px = +0.5×).
   961→      const sensitivity = 100 * previewScale;
   962→      const delta = signedDiag / sensitivity;
   963→      const next = Math.max(0.01, r.startSize + delta);
   964→      onSizeChange(slot, next);
   965→    };
   966→    const onUp = () => {
   967→      resizeRef.current = null;
   968→      window.removeEventListener("mousemove", onMove);
   969→      window.removeEventListener("mouseup", onUp);
   970→    };
   971→    window.addEventListener("mousemove", onMove);
   972→    window.addEventListener("mouseup", onUp);
   973→  }
   974→
   975→  return (
   976→    <div
   977→      ref={containerRef} id={`editable-img-${slotKey(slot)}`}
   978→      className={`${containerClass} group`}
   979→      style={{
   980→        cursor: editable ? "grab" : "default",
   981→        outline: editable ? "2px dashed rgba(0, 102, 255, 0.7)" : undefined,
   982→        outlineOffset: editable ? "-2px" : undefined,
   983→      }}
   984→      onMouseDown={handleMouseDown}
   985→      onDoubleClick={handleDoubleClick}
   986→    >
   987→      <Image
   988→        src={src}
   989→        alt={alt}
   990→        fill
   991→        unoptimized
   992→        className={objectFit === "cover" ? "object-cover" : "object-contain"}
   993→        sizes="700px"
   994→        style={{
   995→          objectPosition: `${focusX}% ${focusY}%`,
   996→          // Apply a tiny overscan (1.005x) on top of the user's zoom to
   997→          // eliminate the 1px white gap that appears at the container
   998→          // edge due to subpixel rendering. This is the well-known
   999→          // "CSS transform scale(1) shows hairline gap" bug — adding a
  1000→          // 0.5% overscan forces the image to spill 1-2px past each
  1001→          // edge, which the parent's overflow:hidden then clips cleanly.
  1002→          transform: `scale(${zoom * 1.005})`,
  1003→          transformOrigin: "center center",
  1004→          // Force GPU compositing so the transform is applied on a
  1005→          // separate layer — eliminates the闪烁 / shimmer that can
  1006→          // happen during drag-pan on Chrome.
  1007→          willChange: "transform",
  1008→          backfaceVisibility: "hidden",
  1009→          transition: dragRef.current ? "none" : "transform 80ms ease-out",
  1010→        }}
  1011→        draggable={false}
  1012→      />
  1013→      {/* Replace button (only in edit mode) */}
  1014→      {editable && onPickImage && (
  1015→        <button
  1016→          type="button"
  1017→          onClick={(e) => {
  1018→            e.stopPropagation();
  1019→            onPickImage(slot);
  1020→          }}
  1021→          className="absolute top-1 left-1 z-10 inline-flex items-center gap-1 rounded bg-[#0066FF] text-white px-2 py-1 text-[10px] font-bold uppercase tracking-wider shadow-md hover:bg-[#0052CC] opacity-0 group-hover:opacity-100 transition"
  1022→          style={{ pointerEvents: "auto" }}
  1023→        >
  1024→          Replace
  1025→        </button>
  1026→      )}
  1027→      {/* Placement readout (only in edit mode) */}
  1028→      {editable && (
  1029→        <div className="absolute bottom-1 right-1 z-10 rounded bg-black/70 px-1.5 py-0.5 text-[9px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none">
  1030→          {Math.round(focusX)}/{Math.round(focusY)} · {zoom.toFixed(1)}×
  1031→        </div>
  1032→      )}
  1033→      {/* Resize corner handles (only when size-control is enabled) */}
  1034→      {editable && onSizeChange && (
  1035→        <>
  1036→          {/* Size readout (top-center pill) */}
  1037→          <div
  1038→            className="absolute top-1 left-1/2 -translate-x-1/2 z-20 rounded bg-[#FF005A] px-2 py-0.5 text-[9px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none whitespace-nowrap"
  1039→          >
  1040→            {sizeLabel ?? "size"}: {(sizeMultiplier ?? 1).toFixed(2)}×
  1041→          </div>
  1042→          {/* NW corner */}
  1043→          <ResizeHandle corner="nw" onMouseDown={handleResizeMouseDown} />
  1044→          {/* NE corner */}
  1045→          <ResizeHandle corner="ne" onMouseDown={handleResizeMouseDown} />
  1046→          {/* SE corner */}
  1047→          <ResizeHandle corner="se" onMouseDown={handleResizeMouseDown} />
  1048→          {/* SW corner */}
  1049→          <ResizeHandle corner="sw" onMouseDown={handleResizeMouseDown} />
  1050→        </>
  1051→      )}
  1052→    </div>
  1053→  );
  1054→}

  1056→/**
  1057→ * ResizeHandle — a small square handle at one of the 4 corners of an
  1058→ * editable image. Dragging it resizes the image via onSizeChange.
  1059→ *
  1060→ * The handle is a 12×12 white square with a 2px pink border, positioned
  1061→ * absolutely at the corner. The cursor changes based on the corner
  1062→ * (nwse or nesw resize cursor).

  1064→ * Visible only in edit mode (the parent conditionally renders it).

  1066→function ResizeHandle({
  1067→  corner,
  1068→  onMouseDown,
  1069→}: {

  1074→    corner === "nw" ? "top-0 left-0" :
  1075→    corner === "ne" ? "top-0 right-0" :
  1076→    corner === "se" ? "bottom-0 right-0" :
  1077→    "bottom-0 left-0";
  1078→  const cursorClass =
  1079→    corner === "nw" || corner === "se" ? "cursor-nwse-resize" : "cursor-nesw-resize";
  1080→  return (
  1081→    <div
  1082→      onMouseDown={(e) => onMouseDown(e, corner)}
  1083→      className={`absolute ${positionClass} ${cursorClass} z-30 w-3 h-3 bg-white border-2 border-[#FF005A] rounded-sm shadow-md opacity-0 group-hover:opacity-100 transition`}
  1084→      style={{ pointerEvents: "auto" }}
  1085→      aria-label={`Resize ${corner} corner`}
  1086→    />
  1087→  );
  1088→}

  1220→/**
  1221→ * SponsorLogo — one logo in the "In collaboration with" / "Sponsored by" row.
  1222→ * Logos use object-contain (no crop), so they don't take a placement.
  1223→ * logoSize: 1 = 32px height (default), 2 = 64px, 0.5 = 16px.
  1224→ */
  1225→function SponsorLogo({
  1226→  sponsor,
  1227→  editable,
  1228→  slot,
  1229→  onPickImage,
  1230→  onSizeChange,
  1231→  previewScale = 1,
  1232→}: {
  1233→  sponsor: { name: string; logoUrl: string; logoSize?: number };
  1234→  editable?: boolean;
  1235→  slot: ImageSlot;
  1236→  onPickImage?: (slot: ImageSlot) => void;
  1237→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
  1238→  previewScale?: number;
  1239→}) {
  1240→  const sizeMult = Math.max(0.01, sponsor.logoSize ?? 1);
  1241→  const heightPx = Math.round(32 * sizeMult);
  1242→  const minWidthPx = Math.round(80 * sizeMult);
  1243→
  1244→  // Resize drag state — same pattern as EditableImage but inline since
  1245→  // SponsorLogo doesn't use EditableImage (logos use object-contain, no
  1246→  // placement control).
  1247→  const resizeRef = useRef<{
  1248→    startX: number;
  1249→    startY: number;
  1250→    startSize: number;
  1251→    corner: "nw" | "ne" | "se" | "sw";
  1252→  } | null>(null);
  1253→
  1254→  function handleResizeMouseDown(
  1255→    e: React.MouseEvent,
  1256→    corner: "nw" | "ne" | "se" | "sw",
  1257→  ) {
  1258→    if (!editable || !onSizeChange) return;
  1259→    if (e.button !== 0) return;
  1260→    e.preventDefault();
  1261→    e.stopPropagation();
  1262→    const startSize = sponsor.logoSize ?? 1;
  1263→    resizeRef.current = { startX: e.clientX, startY: e.clientY, startSize, corner };
  1264→    const onMove = (ev: MouseEvent) => {
  1265→      const r = resizeRef.current;
  1266→      if (!r) return;
  1267→      const dx = ev.clientX - r.startX;
  1268→      const dy = ev.clientY - r.startY;
  1269→      let signedDiag: number;
  1270→      switch (r.corner) {
  1271→        case "se": signedDiag = dx + dy; break;
  1272→        case "nw": signedDiag = -(dx + dy); break;
  1273→        case "ne": signedDiag = dx - dy; break;     // up-right grows (dx>0 grows, dy<0 grows)
  1274→        case "sw": signedDiag = -dx + dy; break;    // down-left grows (dx<0 grows, dy>0 grows)
  1275→      }
  1276→      const sensitivity = 100 * previewScale;
  1277→      const delta = signedDiag / sensitivity;
  1278→      const next = Math.max(0.01, r.startSize + delta);
  1279→      onSizeChange(slot, next);
  1280→    };
  1281→    const onUp = () => {
  1282→      resizeRef.current = null;
  1283→      window.removeEventListener("mousemove", onMove);
  1284→      window.removeEventListener("mouseup", onUp);
  1285→    };
  1286→    window.addEventListener("mousemove", onMove);
  1287→    window.addEventListener("mouseup", onUp);
  1288→  }
  1289→
  1290→  return (
  1291→    <div
  1292→      className={`relative flex items-center justify-center bg-white rounded px-2 py-1 border group ${
  1293→        editable ? "border-[#0066FF]/70" : "border-black/10"
  1294→      }`}
  1295→      style={{ height: `${heightPx}px`, minWidth: `${minWidthPx}px` }}
  1296→    >
  1297→      <div className="relative w-full h-full">
  1298→        <Image
  1299→          src={sponsor.logoUrl}
  1300→          alt={sponsor.name}
  1301→          fill
  1302→          unoptimized
  1303→          className="object-contain"
  1304→          sizes="80px"
  1305→          draggable={false}
  1306→        />
  1307→      </div>
  1308→      {editable && onPickImage && (
  1309→        <button
  1310→          type="button"
  1311→          onClick={(e) => {
  1312→            e.stopPropagation();
  1313→            onPickImage(slot);
  1314→          }}
  1315→          className="absolute -top-1.5 -right-1.5 z-10 rounded-full bg-[#0066FF] text-white px-1.5 py-0.5 text-[9px] font-bold uppercase shadow hover:bg-[#0052CC]"
  1316→        >
  1317→          ↻
  1318→        </button>
  1319→      )}
  1320→      {/* Resize corner handles (only when size-control is enabled) */}
  1321→      {editable && onSizeChange && (
  1322→        <>
  1323→          <div className="absolute -top-4 left-1/2 -translate-x-1/2 z-20 rounded bg-[#FF005A] px-1.5 py-0.5 text-[8px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none whitespace-nowrap">
  1324→            logo: {sizeMult.toFixed(2)}×
  1325→          </div>
  1326→          <div
  1327→            onMouseDown={(e) => handleResizeMouseDown(e, "nw")}
  1328→            className="absolute top-0 left-0 cursor-nwse-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1329→            style={{ pointerEvents: "auto" }}
  1330→          />
  1331→          <div
  1332→            onMouseDown={(e) => handleResizeMouseDown(e, "ne")}
  1333→            className="absolute top-0 right-0 cursor-nesw-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1334→            style={{ pointerEvents: "auto" }}
  1335→          />
  1336→          <div
  1337→            onMouseDown={(e) => handleResizeMouseDown(e, "se")}
  1338→            className="absolute bottom-0 right-0 cursor-nwse-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1339→            style={{ pointerEvents: "auto" }}
  1340→          />
  1341→          <div
  1342→            onMouseDown={(e) => handleResizeMouseDown(e, "sw")}
  1343→            className="absolute bottom-0 left-0 cursor-nesw-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1344→            style={{ pointerEvents: "auto" }}
  1345→          />
  1346→        </>
  1347→      )}
  1348→    </div>
  1349→  );
  1350→}

  1352→/**
  1353→ * QrCode — generates a QR code from a URL using the `qrcode` library.
  1354→ * Renders to a <canvas> so it's high-DPI ready.
  1355→ */
  1356→import QRCode from "qrcode";
  1357→// `useState` and `useEffect` are imported at the top of this file (line 3)
  1358→// — no need to re-import here.
  1359→
  1360→function QrCode({ url, size }: { url: string; size: number }) {
  1361→  const [dataUrl, setDataUrl] = useState<string>("");
  1362→
  1363→  useEffect(() => {
  1364→    let cancelled = false;
  1365→    QRCode.toDataURL(url, {
  1366→      width: size,
  1367→      margin: 0,
  1368→      color: { dark: "#000000", light: "#FFFFFF" },
  1369→      errorCorrectionLevel: "M",
  1370→    })
  1371→      .then((d) => {
  1372→        if (!cancelled) setDataUrl(d);
  1373→      })
  1374→      .catch((err) => {
  1375→        console.error("QR generation failed:", err);
  1376→      });
  1377→    return () => {
  1378→      cancelled = true;
  1379→    };
  1380→  }, [url, size]);
  1381→
  1382→  if (!dataUrl) {
  1383→    return (
  1384→      <div
  1385→        className="bg-black/5 animate-pulse"
  1386→        style={{ width: size, height: size }}
  1387→      />
  1388→    );
  1389→  }
  1390→  // eslint-disable-next-line @next/next/no-img-element
  1391→  return (
  1392→    <img
  1393→      src={dataUrl}
  1394→      alt="QR code"
  1395→      width={size}
  1396→      height={size}
  1397→      className="block"
  1398→    />
  1399→  );
  1400→}
