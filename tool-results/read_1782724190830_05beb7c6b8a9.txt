    24→/**
    25→ * SpeakerIntroCanvas — the data-driven mockup renderer.
    26→ *
    27→ * Takes a `SpeakerIntroData` object and renders the full Speaker Intro
    28→ * mockup as a 3:2 landscape canvas. Every component is a separate
    29→ * absolutely-positioned div bound to a field in `data`. Edit the JSON
    30→ * in the parent editor and the canvas re-renders live.
    31→ *
    32→ * Canvas size: 1200×800 (3:2). The parent scales it down via CSS
    33→ * transform for the on-screen preview; the underlying DOM stays at
    34→ * 1200×800 so PNG export is print-quality.
    35→ *
    36→ * Editable mode (editable=true):
    37→ *   - Image areas show a dashed blue outline + a "Replace" button.
    38→ *   - Click the button → opens the image picker (onPickImage).
    39→ *   - Drag the image → pans (updates focusX/focusY via onPlacementChange).
    40→ *   - Wheel on the image → zooms (updates zoom).
    41→ *   - Double-click → resets placement to default.
    42→ *
    43→ * Sections-editable mode (sectionsEditable=true):
    44→ *   - Text sections (header, topic, speakers, sponsors, collaborators,
    45→ *     branding, qr, footer) get wrapped in <SectionBox> which makes
    46→ *     them draggable + 8-handle resizeable.
    47→ *   - Layout persists in `data.sectionLayout[id] = { pos, scale }`.
    48→ *   - Alignment guides appear when dragging (cyan lines at canvas edges,
    49→ *     centers, and peer box edges).
    50→ *   - Text sections always render at zIndex >= 50 so they stay above
    51→ *     images and overlays.
    52→ */
    53→
    54→const CANVAS_W = 1200;
    55→const CANVAS_H = 800;
    56→
    57→type Props = {
    58→  data: SpeakerIntroData;
    59→  className?: string;
    60→  /** When true, image areas become interactive (drag/wheel/click). */
    61→  editable?: boolean;
    62→  /** When true, text sections become draggable + resizeable. */
    63→  sectionsEditable?: boolean;
    64→  /** Called when the user clicks "Replace" on an image slot. */
    65→  onPickImage?: (slot: ImageSlot) => void;
    66→  /** Called whenever an image is dragged / zoomed. */
    67→  onPlacementChange?: (slot: ImageSlot, placement: ImagePlacement) => void;
    68→  /** Called when the user drags a resize corner handle on an image. */
    69→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
    70→  /** Called when a section is dragged to a new position. */
    71→  onSectionMove?: (id: SectionId, pos: SectionPos) => void;
    72→  /** Called when a section is resized via a corner/edge handle. */
    73→  onSectionResize?: (id: SectionId, scale: number) => void;
    74→  /** Called when a section is resized via a mid-edge handle — updates the
    75→   *  box's explicit width/height in canvas px. */
    76→  onSectionBoxResize?: (id: SectionId, size: SectionBoxSize) => void;
    77→  /** Called when the hero overlay z-index changes (front/back button). */
    78→  onHeroZChange?: (z: number) => void;
    79→  /** Called when the triangle overlay z-index changes (front/back button). */
    80→  onTriangleZChange?: (z: number) => void;
    81→  /** Called when the hero overlay X scale changes (slider). */
    82→  onHeroScaleXChange?: (n: number) => void;
    83→  /** Called when the hero overlay Y scale changes (slider). */
    84→  onHeroScaleYChange?: (n: number) => void;
    85→  /** Called when a section's z-index changes (Front/Back in ObjectPropertiesPanel). */
    86→  onSectionZChange?: (id: SectionId, z: number) => void;
    87→  /** The current scale of the preview (used to convert screen-drag to canvas-%). */
    88→  previewScale?: number;
    89→};

   113→    // Default z-index for hero / triangle / text layers.

   199→          const scaleY = Math.max(0.01, data.heroOverlay.imageScaleY ?? 1);

   205→          style={{
   206→            left: `${heroLeft}%`,
   207→            top: `${heroTop}%`,
   208→            width: `${heroWidth}%`,

   274→                      key={i}

   287→            </div>

   290→          {/* 7. LOCATION PINS — overlay with connector lines */}

   292→            className="absolute inset-0 w-full h-full pointer-events-none"

   361→          label="Header"

   686→/**
   687→ * EditableImage — wraps a next/image with placement (object-position + scale)
   688→ * and (optionally) edit-mode interactions: click-to-replace, drag-to-pan,
   689→ * wheel-to-zoom, double-click-to-reset.
   690→ *
   691→ * Used for the hero background and the speaker headshots.
   692→ */
   693→function EditableImage({
   694→  slot,
   695→  src,
   696→  alt,
   697→  placement,
   698→  editable,
   699→  previewScale,
   700→  onPickImage,
   701→  onPlacementChange,
   702→  onSizeChange,
   703→  sizeMultiplier,
   704→  sizeLabel,
   705→  containerClass,
   706→  objectFit,
   707→}: {
   708→  slot: ImageSlot;
   709→  src: string;
   710→  alt: string;
   711→  placement?: ImagePlacement;
   712→  editable?: boolean;
   713→  previewScale: number;
   714→  onPickImage?: (slot: ImageSlot) => void;
   715→  onPlacementChange?: (slot: ImageSlot, p: ImagePlacement) => void;
   716→  /** Called when the user drags a resize corner handle. Receives the new
   717→   *  size multiplier (e.g. 1.5 = 150% of default). */
   718→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
   719→  /** Current size multiplier (read from data). Used to seed the resize
   720→   *  drag delta. Defaults to 1. */
   721→  sizeMultiplier?: number;
   722→  /** Small label shown next to the resize readout (e.g. "photo", "logo"). */
   723→  sizeLabel?: string;
   724→  containerClass: string;
   725→  objectFit: "cover" | "contain";
   726→}) {
   727→  const { focusX, focusY, zoom } = resolvePlacement(placement);
   728→  // We track drag state on a ref so we don't re-render on every mousemove.
   729→  const dragRef = useRef<{
   730→    startX: number;
   731→    startY: number;
   732→    startFocusX: number;
   733→    startFocusY: number;
   734→  } | null>(null);
   735→  // Resize drag state — separate from pan drag so they don't conflict.
   736→  const resizeRef = useRef<{
   737→    startX: number;
   738→    startY: number;
   739→    startSize: number;
   740→    /** Which corner is being dragged. The corner determines whether
   741→     *  moving the mouse up-left grows or shrinks the image. */
   742→    corner: "nw" | "ne" | "se" | "sw";
   743→  } | null>(null);
   744→
   745→  function handleMouseDown(e: React.MouseEvent) {
   746→    if (!editable || !onPlacementChange) return;
   747→    // Only start a drag on left-click outside the Replace button.
   748→    if (e.button !== 0) return;
   749→    e.preventDefault();
   750→    dragRef.current = {
   751→      startX: e.clientX,
   752→      startY: e.clientY,
   753→      startFocusX: focusX,
   754→      startFocusY: focusY,
   755→    };
   756→    // Switch to grabbing cursor.
   757→    (e.currentTarget as HTMLElement).style.cursor = "grabbing";
   758→
   759→    const onMove = (ev: MouseEvent) => {
   760→      const d = dragRef.current;
   761→      if (!d) return;
   762→      const dx = ev.clientX - d.startX;
   763→      const dy = ev.clientY - d.startY;
   764→      // Convert screen-pixel delta to % of container.
   765→      // The container's pixel width at screen-scale is:
   766→      //   containerWidthPx (canvas) * previewScale.
   767→      // We don't know containerWidthPx here without measuring it; use a
   768→      // reasonable approximation: 1% per (previewScale * 6) px, which gives
   769→      // ~6 px of drag = 1% on the canvas at scale=1.
   770→      const sensitivity = 6 * previewScale;
   771→      const nextFocusX = Math.max(
   772→        0,
   773→        Math.min(100, d.startFocusX - dx / sensitivity),
   774→      );
   775→      const nextFocusY = Math.max(
   776→        0,
   777→        Math.min(100, d.startFocusY - dy / sensitivity),
   778→      );
   779→      onPlacementChange(slot, {
   780→        focusX: nextFocusX,
   781→        focusY: nextFocusY,
   782→        zoom,
   783→      });
   784→    };
   785→    const onUp = () => {
   786→      dragRef.current = null;
   787→      window.removeEventListener("mousemove", onMove);
   788→      window.removeEventListener("mouseup", onUp);
   789→      // Restore cursor.
   790→      const el = document.getElementById(`editable-img-${slotKey(slot)}`);
   791→      if (el) el.style.cursor = "grab";
   792→    };
   793→    window.addEventListener("mousemove", onMove);
   794→    window.addEventListener("mouseup", onUp);
   795→  }
   796→
   797→  const containerRef = useRef<HTMLDivElement>(null);
   798→    // Attach a NON-PASSIVE wheel listener so preventDefault
   799→    // actually stops the parent workspace from scrolling.
   800→    // React's onWheel is passive by default → preventDefault
   801→    // is a no-op there + logs a console warning.
   802→    useNonPassiveWheel(containerRef, handleWheel, !!editable);
   803→
   804→    function handleWheel(e: WheelEvent) {
   805→    if (!editable || !onPlacementChange) return;
   806→    // preventDefault + stopPropagation are already called by the
   807→    // useNonPassiveWheel hook (non-passive native listener), so
   808→    // the parent workspace does not scroll while the user spins
   809→    // the wheel over a hovered image.
   810→const step = e.deltaY < 0 ? 0.1 : -0.1;
   811→    const nextZoom = Math.max(0.01, zoom + step);
   812→    onPlacementChange(slot, {
   813→      focusX,
   814→      focusY,
   815→      zoom: nextZoom,
   816→    });
   817→  }
   818→
   819→  function handleDoubleClick() {
   820→    if (!editable || !onPlacementChange) return;
   821→    onPlacementChange(slot, { focusX: 50, focusY: 50, zoom: 1 });
   822→  }
   823→
   824→  /**
   825→   * handleResizeMouseDown — starts a resize drag from one of the 4 corner
   826→   * handles. The corner determines the direction:
   827→   *   - SE (bottom-right): drag down-right = grow, up-left = shrink
   828→   *   - NW (top-left):     drag up-left   = grow, down-right = shrink
   829→   *   - NE (top-right):    drag up-right  = grow, down-left = shrink
   830→   *   - SW (bottom-left):  drag down-left = grow, up-right = shrink
   831→   *
   832→   * We use the diagonal distance (dx + dy with appropriate sign) so the
   833→   * resize feels natural regardless of which corner is grabbed.
   834→   */
   835→  function handleResizeMouseDown(
   836→    e: React.MouseEvent,
   837→    corner: "nw" | "ne" | "se" | "sw",
   838→  ) {
   839→    if (!editable || !onSizeChange) return;
   840→    if (e.button !== 0) return;
   841→    e.preventDefault();
   842→    e.stopPropagation(); // don't trigger pan-drag
   843→    const startSize = sizeMultiplier ?? 1;
   844→    resizeRef.current = { startX: e.clientX, startY: e.clientY, startSize, corner };
   845→
   846→    const onMove = (ev: MouseEvent) => {
   847→      const r = resizeRef.current;
   848→      if (!r) return;
   849→      const dx = ev.clientX - r.startX;
   850→      const dy = ev.clientY - r.startY;
   851→      // For each corner, compute the signed diagonal distance so that
   852→      // moving "outward" (away from the image center) increases the size
   853→      // and moving "toward the center" decreases it. The SW and NE
   854→      // formulas were previously inverted — fixed per user spec.
   855→      let signedDiag: number;
   856→      switch (r.corner) {
   857→        case "se": signedDiag = dx + dy; break;            // down-right grows
   858→        case "nw": signedDiag = -(dx + dy); break;          // up-left grows
   859→        case "ne": signedDiag = dx - dy; break;             // up-right grows (dx>0 grows, dy<0 grows)
   860→        case "sw": signedDiag = -dx + dy; break;            // down-left grows (dx<0 grows, dy>0 grows)
   861→      }
   862→      // 100px of drag = 1.0× size change (so dragging 50px = +0.5×).
   863→      const sensitivity = 100 * previewScale;
   864→      const delta = signedDiag / sensitivity;
   865→      const next = Math.max(0.01, r.startSize + delta);
   866→      onSizeChange(slot, next);
   867→    };
   868→    const onUp = () => {
   869→      resizeRef.current = null;
   870→      window.removeEventListener("mousemove", onMove);
   871→      window.removeEventListener("mouseup", onUp);
   872→    };
   873→    window.addEventListener("mousemove", onMove);
   874→    window.addEventListener("mouseup", onUp);
   875→  }
   876→
   877→  return (
   878→    <div
   879→      ref={containerRef} id={`editable-img-${slotKey(slot)}`}
   880→      className={`${containerClass} group`}
   881→      style={{
   882→        cursor: editable ? "grab" : "default",
   883→        outline: editable ? "2px dashed rgba(0, 102, 255, 0.7)" : undefined,
   884→        outlineOffset: editable ? "-2px" : undefined,
   885→      }}
   886→      onMouseDown={handleMouseDown}
   887→      onDoubleClick={handleDoubleClick}
   888→    >
   889→      <Image
   890→        src={src}
   891→        alt={alt}
   892→        fill
   893→        unoptimized
   894→        className={objectFit === "cover" ? "object-cover" : "object-contain"}
   895→        sizes="700px"
   896→        style={{
   897→          objectPosition: `${focusX}% ${focusY}%`,
   898→          // Apply a tiny overscan (1.005x) on top of the user's zoom to
   899→          // eliminate the 1px white gap that appears at the container
   900→          // edge due to subpixel rendering. This is the well-known
   901→          // "CSS transform scale(1) shows hairline gap" bug — adding a
   902→          // 0.5% overscan forces the image to spill 1-2px past each
   903→          // edge, which the parent's overflow:hidden then clips cleanly.
   904→          transform: `scale(${zoom * 1.005})`,
   905→          transformOrigin: "center center",
   906→          // Force GPU compositing so the transform is applied on a
   907→          // separate layer — eliminates the闪烁 / shimmer that can
   908→          // happen during drag-pan on Chrome.
   909→          willChange: "transform",
   910→          backfaceVisibility: "hidden",
   911→          transition: dragRef.current ? "none" : "transform 80ms ease-out",
   912→        }}
   913→        draggable={false}
   914→      />
   915→      {/* Replace button (only in edit mode) */}
   916→      {editable && onPickImage && (
   917→        <button
   918→          type="button"
   919→          onClick={(e) => {
   920→            e.stopPropagation();
   921→            onPickImage(slot);
   922→          }}
   923→          className="absolute top-1 left-1 z-10 inline-flex items-center gap-1 rounded bg-[#0066FF] text-white px-2 py-1 text-[10px] font-bold uppercase tracking-wider shadow-md hover:bg-[#0052CC] opacity-0 group-hover:opacity-100 transition"
   924→          style={{ pointerEvents: "auto" }}
   925→        >
   926→          Replace
   927→        </button>
   928→      )}
   929→      {/* Placement readout (only in edit mode) */}
   930→      {editable && (
   931→        <div className="absolute bottom-1 right-1 z-10 rounded bg-black/70 px-1.5 py-0.5 text-[9px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none">
   932→          {Math.round(focusX)}/{Math.round(focusY)} · {zoom.toFixed(1)}×
   933→        </div>
   934→      )}
   935→      {/* Resize corner handles (only when size-control is enabled) */}
   936→      {editable && onSizeChange && (
   937→        <>
   938→          {/* Size readout (top-center pill) */}
   939→          <div
   940→            className="absolute top-1 left-1/2 -translate-x-1/2 z-20 rounded bg-[#FF005A] px-2 py-0.5 text-[9px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none whitespace-nowrap"
   941→          >
   942→            {sizeLabel ?? "size"}: {(sizeMultiplier ?? 1).toFixed(2)}×
   943→          </div>
   944→          {/* NW corner */}
   945→          <ResizeHandle corner="nw" onMouseDown={handleResizeMouseDown} />
   946→          {/* NE corner */}
   947→          <ResizeHandle corner="ne" onMouseDown={handleResizeMouseDown} />
   948→          {/* SE corner */}
   949→          <ResizeHandle corner="se" onMouseDown={handleResizeMouseDown} />
   950→          {/* SW corner */}
   951→          <ResizeHandle corner="sw" onMouseDown={handleResizeMouseDown} />
   952→        </>
   953→      )}
   954→    </div>
   955→  );
   956→}

   958→/**
   959→ * ResizeHandle — a small square handle at one of the 4 corners of an
   960→ * editable image. Dragging it resizes the image via onSizeChange.
   961→ *
   962→ * The handle is a 12×12 white square with a 2px pink border, positioned
   963→ * absolutely at the corner. The cursor changes based on the corner
   964→ * (nwse or nesw resize cursor).
   965→ *
   966→ * Visible only in edit mode (the parent conditionally renders it).
   967→ */
   968→function ResizeHandle({
   969→  corner,
   970→  onMouseDown,
   971→}: {
   972→  corner: "nw" | "ne" | "se" | "sw";
   973→  onMouseDown: (e: React.MouseEvent, corner: "nw" | "ne" | "se" | "sw") => void;
   974→}) {
   975→  const positionClass =
   976→    corner === "nw" ? "top-0 left-0" :
   977→    corner === "ne" ? "top-0 right-0" :
   978→    corner === "se" ? "bottom-0 right-0" :
   979→    "bottom-0 left-0";
   980→  const cursorClass =
   981→    corner === "nw" || corner === "se" ? "cursor-nwse-resize" : "cursor-nesw-resize";
   982→  return (
   983→    <div
   984→      onMouseDown={(e) => onMouseDown(e, corner)}
   985→      className={`absolute ${positionClass} ${cursorClass} z-30 w-3 h-3 bg-white border-2 border-[#FF005A] rounded-sm shadow-md opacity-0 group-hover:opacity-100 transition`}
   986→      style={{ pointerEvents: "auto" }}
   987→      aria-label={`Resize ${corner} corner`}
   988→    />
   989→  );
   990→}

  1122→/**
  1123→ * SponsorLogo — one logo in the "In collaboration with" / "Sponsored by" row.
  1124→ * Logos use object-contain (no crop), so they don't take a placement.
  1125→ * logoSize: 1 = 32px height (default), 2 = 64px, 0.5 = 16px.
  1126→ */
  1127→function SponsorLogo({
  1128→  sponsor,
  1129→  editable,
  1130→  slot,
  1131→  onPickImage,
  1132→  onSizeChange,
  1133→  previewScale = 1,
  1134→}: {
  1135→  sponsor: { name: string; logoUrl: string; logoSize?: number };
  1136→  editable?: boolean;
  1137→  slot: ImageSlot;
  1138→  onPickImage?: (slot: ImageSlot) => void;
  1139→  onSizeChange?: (slot: ImageSlot, newMultiplier: number) => void;
  1140→  previewScale?: number;
  1141→}) {
  1142→  const sizeMult = Math.max(0.01, sponsor.logoSize ?? 1);
  1143→  const heightPx = Math.round(32 * sizeMult);
  1144→  const minWidthPx = Math.round(80 * sizeMult);
  1145→
  1146→  // Resize drag state — same pattern as EditableImage but inline since
  1147→  // SponsorLogo doesn't use EditableImage (logos use object-contain, no
  1148→  // placement control).
  1149→  const resizeRef = useRef<{
  1150→    startX: number;
  1151→    startY: number;
  1152→    startSize: number;
  1153→    corner: "nw" | "ne" | "se" | "sw";
  1154→  } | null>(null);
  1155→
  1156→  function handleResizeMouseDown(
  1157→    e: React.MouseEvent,
  1158→    corner: "nw" | "ne" | "se" | "sw",
  1159→  ) {
  1160→    if (!editable || !onSizeChange) return;
  1161→    if (e.button !== 0) return;
  1162→    e.preventDefault();
  1163→    e.stopPropagation();
  1164→    const startSize = sponsor.logoSize ?? 1;
  1165→    resizeRef.current = { startX: e.clientX, startY: e.clientY, startSize, corner };
  1166→    const onMove = (ev: MouseEvent) => {
  1167→      const r = resizeRef.current;
  1168→      if (!r) return;
  1169→      const dx = ev.clientX - r.startX;
  1170→      const dy = ev.clientY - r.startY;
  1171→      let signedDiag: number;
  1172→      switch (r.corner) {
  1173→        case "se": signedDiag = dx + dy; break;
  1174→        case "nw": signedDiag = -(dx + dy); break;
  1175→        case "ne": signedDiag = dx - dy; break;     // up-right grows (dx>0 grows, dy<0 grows)
  1176→        case "sw": signedDiag = -dx + dy; break;    // down-left grows (dx<0 grows, dy>0 grows)
  1177→      }
  1178→      const sensitivity = 100 * previewScale;
  1179→      const delta = signedDiag / sensitivity;
  1180→      const next = Math.max(0.01, r.startSize + delta);
  1181→      onSizeChange(slot, next);
  1182→    };
  1183→    const onUp = () => {
  1184→      resizeRef.current = null;
  1185→      window.removeEventListener("mousemove", onMove);
  1186→      window.removeEventListener("mouseup", onUp);
  1187→    };
  1188→    window.addEventListener("mousemove", onMove);
  1189→    window.addEventListener("mouseup", onUp);
  1190→  }
  1191→
  1192→  return (
  1193→    <div
  1194→      className={`relative flex items-center justify-center bg-white rounded px-2 py-1 border group ${
  1195→        editable ? "border-[#0066FF]/70" : "border-black/10"
  1196→      }`}
  1197→      style={{ height: `${heightPx}px`, minWidth: `${minWidthPx}px` }}
  1198→    >
  1199→      <div className="relative w-full h-full">
  1200→        <Image
  1201→          src={sponsor.logoUrl}
  1202→          alt={sponsor.name}
  1203→          fill
  1204→          unoptimized
  1205→          className="object-contain"
  1206→          sizes="80px"
  1207→          draggable={false}
  1208→        />
  1209→      </div>
  1210→      {editable && onPickImage && (
  1211→        <button
  1212→          type="button"
  1213→          onClick={(e) => {
  1214→            e.stopPropagation();
  1215→            onPickImage(slot);
  1216→          }}
  1217→          className="absolute -top-1.5 -right-1.5 z-10 rounded-full bg-[#0066FF] text-white px-1.5 py-0.5 text-[9px] font-bold uppercase shadow hover:bg-[#0052CC]"
  1218→        >
  1219→          ↻
  1220→        </button>
  1221→      )}
  1222→      {/* Resize corner handles (only when size-control is enabled) */}
  1223→      {editable && onSizeChange && (
  1224→        <>
  1225→          <div className="absolute -top-4 left-1/2 -translate-x-1/2 z-20 rounded bg-[#FF005A] px-1.5 py-0.5 text-[8px] font-mono text-white opacity-0 group-hover:opacity-100 transition pointer-events-none whitespace-nowrap">
  1226→            logo: {sizeMult.toFixed(2)}×
  1227→          </div>
  1228→          <div
  1229→            onMouseDown={(e) => handleResizeMouseDown(e, "nw")}
  1230→            className="absolute top-0 left-0 cursor-nwse-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1231→            style={{ pointerEvents: "auto" }}
  1232→          />
  1233→          <div
  1234→            onMouseDown={(e) => handleResizeMouseDown(e, "ne")}
  1235→            className="absolute top-0 right-0 cursor-nesw-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1236→            style={{ pointerEvents: "auto" }}
  1237→          />
  1238→          <div
  1239→            onMouseDown={(e) => handleResizeMouseDown(e, "se")}
  1240→            className="absolute bottom-0 right-0 cursor-nwse-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1241→            style={{ pointerEvents: "auto" }}
  1242→          />
  1243→          <div
  1244→            onMouseDown={(e) => handleResizeMouseDown(e, "sw")}
  1245→            className="absolute bottom-0 left-0 cursor-nesw-resize z-30 w-2.5 h-2.5 bg-white border-2 border-[#FF005A] rounded-sm shadow opacity-0 group-hover:opacity-100 transition"
  1246→            style={{ pointerEvents: "auto" }}
  1247→          />
  1248→        </>
  1249→      )}
  1250→    </div>
  1251→  );
  1252→}

  1254→/**
  1255→ * QrCode — generates a QR code from a URL using the `qrcode` library.
  1256→ * Renders to a <canvas> so it's high-DPI ready.
  1257→ */
  1258→import QRCode from "qrcode";
  1259→// `useState` and `useEffect` are imported at the top of this file (line 3)
  1260→// — no need to re-import here.
  1261→
  1262→function QrCode({ url, size }: { url: string; size: number }) {
  1263→  const [dataUrl, setDataUrl] = useState<string>("");
  1264→
  1265→  useEffect(() => {
  1266→    let cancelled = false;
  1267→    QRCode.toDataURL(url, {
  1268→      width: size,
  1269→      margin: 0,
  1270→      color: { dark: "#000000", light: "#FFFFFF" },
  1271→      errorCorrectionLevel: "M",
  1272→    })
  1273→      .then((d) => {
  1274→        if (!cancelled) setDataUrl(d);
  1275→      })
  1276→      .catch((err) => {
  1277→        console.error("QR generation failed:", err);
  1278→      });
  1279→    return () => {
  1280→      cancelled = true;
  1281→    };
  1282→  }, [url, size]);
  1283→
  1284→  if (!dataUrl) {
  1285→    return (
  1286→      <div
  1287→        className="bg-black/5 animate-pulse"
  1288→        style={{ width: size, height: size }}
  1289→      />
  1290→    );
  1291→  }
  1292→  // eslint-disable-next-line @next/next/no-img-element
  1293→  return (
  1294→    <img
  1295→      src={dataUrl}
  1296→      alt="QR code"
  1297→      width={size}
  1298→      height={size}
  1299→      className="block"
  1300→    />
  1301→  );
  1302→}
