    36→/**
    37→ * SpeakerIntroEditor — the editor + live preview surface.
    38→ *
    39→ * Layout:
    40→ *   - Top toolbar: Event picker / Edit-mode toggle / JSON toggle / Reset / Copy / Download
    41→ *   - Left panel: JSON editor (textarea-based, syntax-ish)
    42→ *   - Right panel: Live mockup preview (scaled to fit) + error overlay
    43→ *
    44→ * Workflows:
    45→ *   1. Pick an event from the dropdown → fetch /api/events/[slug] →
    46→ *      mapEventToSpeakerIntroData → data is replaced.
    47→ *   2. Toggle "Edit images" → image areas on the canvas become
    48→ *      interactive. Click "Replace" to open the image picker.
    49→ *      Drag to pan. Wheel to zoom. Double-click to reset.
    50→ *   3. Edit JSON directly → canvas re-renders live.
    51→ */
    52→
    53→const STORAGE_KEY = "speaker-intro-data-v1";
    54→
    55→type Props = {
    56→  /** Lightweight event list for the dropdown (passed from server). */
    57→  events: EventPickListItem[];
    58→};

    60→export function SpeakerIntroEditor({ events }: Props) {
    61→  const [data, setData] = useState<SpeakerIntroData>(SAMPLE_DATA);
    62→  const [jsonText, setJsonText] = useState<string>(() =>
    63→    JSON.stringify(SAMPLE_DATA, null, 2),
    64→  );
    65→  const [parseError, setParseError] = useState<string | null>(null);
    66→  /** View mode for the left panel: "form" (structured inputs) or "json" (raw textarea). */
    67→  const [viewMode, setViewMode] = useState<"form" | "json">("form");
    68→  const [copied, setCopied] = useState(false);
    69→  const [downloading, setDownloading] = useState(false);
    70→  const [previewScale, setPreviewScale] = useState<number>(0.5);
    71→  /** Edit mode = image areas are interactive (drag/wheel/click). */
    72→  const [editMode, setEditMode] = useState<boolean>(false);
    73→  /** Sections edit mode = text sections are draggable + resizeable. */
    74→  const [sectionsEditMode, setSectionsEditMode] = useState<boolean>(false);
    75→  /** Currently selected event slug in the dropdown. */
    76→  const [selectedEventSlug, setSelectedEventSlug] = useState<string>("");
    77→  /** Loading state when fetching event data. */
    78→  const [loadingEvent, setLoadingEvent] = useState(false);
    79→  /** Image picker modal state. */
    80→  const [pickerSlot, setPickerSlot] = useState<ImageSlot | null>(null);
    81→
    82→  const canvasRef = useRef<HTMLDivElement>(null);
    83→  const previewContainerRef = useRef<HTMLDivElement>(null);
    84→
    85→  // --- helpers ---------------------------------------------------------
    86→
    87→  // rafRef is declared BEFORE applyData so applyData can use it to debounce
    88→  // JSON serialization. Without debouncing, every keystroke in the form
    89→  // triggers a synchronous JSON.stringify — for large data objects this
    90→  // causes noticeable typing lag.
    91→  const rafRef = useRef<number | null>(null);
    92→
    93→  /** Apply a new SpeakerIntroData object: update data + JSON + clear error.
    94→   *  The JSON text update is debounced via requestAnimationFrame so rapid
    95→   *  edits (e.g. typing in a form field) don't trigger a serialize on every
    96→   *  single keystroke. */
    97→  const applyData = useCallback((next: SpeakerIntroData) => {
    98→    setData(next);
    99→    setParseError(null);
   100→    if (rafRef.current === null) {
   101→      rafRef.current = requestAnimationFrame(() => {
   102→        rafRef.current = null;
   103→        setJsonText(JSON.stringify(next, null, 2));
   104→      });
   105→    }
   106→  }, []);
   107→
   108→  /** Update a single image's URL based on its slot. */
   109→  function applyImagePick(slot: ImageSlot, url: string): SpeakerIntroData {
   110→    const next: SpeakerIntroData = JSON.parse(JSON.stringify(data));
   111→    if (slot.kind === "hero") {
   112→      // Auto-hide the triangle overlay when the user picks a new hero image
   113→      // (per user spec). They can re-enable it in the form/JSON editor.
   114→      const prevUrl = next.heroOverlay.imageUrl;
   115→      next.heroOverlay.imageUrl = url;
   116→      if (prevUrl && url && prevUrl !== url) {
   117→        next.heroOverlay.showTriangleOverlay = false;
   118→      }
   119→    } else if (slot.kind === "speaker") {
   120→      const sp = next.speakers.sort((a, b) => a.order - b.order)[slot.index];
   121→      if (sp) sp.photoUrl = url;
   122→    } else {
   123→      const arr = slot.group === "collaborators" ? next.collaborators : next.sponsors;
   124→      const item = arr[slot.index];
   125→      if (item) item.logoUrl = url;
   126→    }
   127→    return next;
   128→  }
   129→
   130→  /** Update a single image's placement based on its slot. */
   131→  function applyPlacementChange(
   132→    slot: ImageSlot,
   133→    placement: ImagePlacement,
   134→  ): SpeakerIntroData {
   135→    const next: SpeakerIntroData = JSON.parse(JSON.stringify(data));
   136→    if (slot.kind === "hero") {
   137→      next.heroOverlay.imagePlacement = placement;
   138→    } else if (slot.kind === "speaker") {
   139→      const sp = next.speakers.sort((a, b) => a.order - b.order)[slot.index];
   140→      if (sp) sp.photoPlacement = placement;
   141→    }
   142→    // sponsors use object-contain — no placement.
   143→    return next;
   144→  }
   145→
   146→  /** Get the current URL for a slot (used to highlight current pick). */
   147→  function urlForSlot(slot: ImageSlot): string | undefined {
   148→    if (slot.kind === "hero") return data.heroOverlay.imageUrl;
   149→    if (slot.kind === "speaker") {
   150→      const sp = [...data.speakers].sort((a, b) => a.order - b.order)[slot.index];
   151→      return sp?.photoUrl;
   152→    }
   153→    const arr = slot.group === "collaborators" ? data.collaborators : data.sponsors;
   154→    return arr[slot.index]?.logoUrl;
   155→  }
   156→
   157→  // --- auto-scale preview ---------------------------------------------
   158→
   159→  useEffect(() => {
   160→    const el = previewContainerRef.current;
   161→    if (!el) return;
   162→    const compute = () => {
   163→      const avail = el.clientWidth - 32;
   164→      const scale = Math.min(1, Math.max(0.2, avail / 1200));
   165→      setPreviewScale(scale);
   166→    };
   167→    compute();
   168→    const ro = new ResizeObserver(compute);
   169→    ro.observe(el);
   170→    return () => ro.disconnect();
   171→  }, [viewMode]);
   172→
   173→  // --- localStorage hydration -----------------------------------------
   174→
   175→  useEffect(() => {
   176→    try {
   177→      const saved = localStorage.getItem(STORAGE_KEY);
   178→      if (saved) {
   179→        const parsed = JSON.parse(saved) as SpeakerIntroData;
   180→        setData(parsed);
   181→        setJsonText(JSON.stringify(parsed, null, 2));
   182→      }
   183→    } catch {
   184→      // ignore — fall back to sample
   185→    }
   186→  }, []);
   187→
   188→  useEffect(() => {
   189→    try {
   190→      localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
   191→    } catch {
   192→      // ignore quota errors
   193→    }
   194→  }, [data]);
   195→
   196→  // --- JSON textarea → data ------------------------------------------
   197→
   198→  const handleJsonChange = useCallback((next: string) => {
   199→    setJsonText(next);
   200→    try {
   201→      const parsed = JSON.parse(next) as SpeakerIntroData;
   202→      if (!parsed || typeof parsed !== "object") {
   203→        throw new Error("Root must be an object");
   204→      }
   205→      if (!parsed.event || !parsed.speakers) {
   206→        throw new Error("Missing required fields: event, speakers");
   207→      }
   208→      setData(parsed);
   209→      setParseError(null);
   210→    } catch (err) {
   211→      setParseError(err instanceof Error ? err.message : "Invalid JSON");
   212→    }
   213→  }, []);
   214→
   215→  // --- event picker ---------------------------------------------------
   216→
   217→  async function handleEventPick(slug: string) {
   218→    setSelectedEventSlug(slug);
   219→    if (!slug) return;
   220→    setLoadingEvent(true);
   221→    setParseError(null);
   222→    try {
   223→      const res = await fetch(`/api/events/${slug}`, { cache: "no-store" });
   224→      if (!res.ok) {
   225→        throw new Error(`Failed to load event (HTTP ${res.status})`);
   226→      }
   227→      const json = (await res.json()) as { event: DbEventForMapping };
   228→      const mapped = mapEventToSpeakerIntroData(json.event);
   229→      applyData(mapped);
   230→    } catch (err) {
   231→      setParseError(err instanceof Error ? err.message : "Failed to load event");
   232→    } finally {
   233→      setLoadingEvent(false);
   234→    }
   235→  }
   236→
   237→  // --- image picker callbacks ----------------------------------------
   238→
   239→  function handlePickImage(slot: ImageSlot) {
   240→    setPickerSlot(slot);
   241→  }
   242→
   243→  function handlePickerSelect(url: string) {
   244→    if (!pickerSlot) return;
   245→    const next = applyImagePick(pickerSlot, url);
   246→    applyData(next);
   247→    setPickerSlot(null);
   248→  }
   249→
   250→  function handlePlacementChange(slot: ImageSlot, placement: ImagePlacement) {
   251→    // Don't call applyData here — it would re-render the JSON textarea on
   252→    // every mousemove (janky). Instead, update data + JSON directly.
   253→    const next = applyPlacementChange(slot, placement);
   254→    setData(next);
   255→    // Debounce the JSON textarea update via requestAnimationFrame.
   256→    if (rafRef.current === null) {
   257→      rafRef.current = requestAnimationFrame(() => {
   258→        rafRef.current = null;
   259→        setJsonText(JSON.stringify(next, null, 2));
   260→      });
   261→    }
   262→  }
   263→
   264→  /** Apply a resize drag to the data — updates the size multiplier on the
   265→   *  targeted slot (imageScale for hero, photoSize for speaker, logoSize
   266→   *  for sponsor). Same debounced-JSON pattern as handlePlacementChange. */
   267→  function applySizeChange(slot: ImageSlot, newMultiplier: number): SpeakerIntroData {
   268→    const next: SpeakerIntroData = JSON.parse(JSON.stringify(data));
   269→    if (slot.kind === "hero") {
   270→      next.heroOverlay.imageScale = newMultiplier;
   271→    } else if (slot.kind === "speaker") {
   272→      const sp = next.speakers.sort((a, b) => a.order - b.order)[slot.index];
   273→      if (sp) sp.photoSize = newMultiplier;
   274→    } else {
   275→      const arr = slot.group === "collaborators" ? next.collaborators : next.sponsors;
   276→      const item = arr[slot.index];
   277→      if (item) item.logoSize = newMultiplier;
   278→    }
   279→    return next;
   280→  }

   421→  async function handleCopyJson() {

   431→  /** Export the canvas to a PNG data URL (used by Share buttons + Download). */
   432→  const getPngDataUrl = useCallback(async (): Promise<string> => {
   433→    if (!canvasRef.current) {
   434→      throw new Error("Canvas not ready");
   435→    }
   436→    return toPng(canvasRef.current, {
   437→      pixelRatio: 2,
   438→      cacheBust: true,
   439→      backgroundColor: "#ffffff",
   440→    });
   441→  }, []);
   442→
   443→  async function handleDownloadPng() {
   444→    if (!canvasRef.current) return;
   445→    setDownloading(true);
   446→    try {
   447→      const dataUrl = await getPngDataUrl();
   448→      const link = document.createElement("a");
   449→      const slug = data.event.name
   450→        .toLowerCase()
   451→        .replace(/[^a-z0-9]+/g, "-")
   452→        .replace(/^-|-$/g, "");
   453→      link.download = `speaker-intro-${slug || "mockup"}-${Date.now()}.png`;
   454→      link.href = dataUrl;
   455→      link.click();
   456→    } catch (err) {
   457→      console.error("PNG export failed:", err);
   458→      alert("PNG export failed — see console for details.");
   459→    } finally {
   460→      setDownloading(false);
   461→    }
   462→  }
